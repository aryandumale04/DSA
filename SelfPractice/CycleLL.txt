 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        // unordered_set <ListNode*> nodes;
        // ListNode *temp = head;
        // while(temp != NULL) {
        //     if(nodes.find(temp) == nodes.end()) {
        //         // node first time hai store it

        //     } else {
        //         //cycle 
        //         return true;
            
        //     }
        //     temp = temp -> next;
        // }
        // return false;



        //constant space approach 
        // Floyd's Cycle Detection Algorithm 
        ListNode* slow = head;
        ListNode* fast = head;
        while(fast != NULL && fast -> next != NULL) {
            slow = slow -> next;
            fast = fast -> next -> next;

            if(slow == fast) return true;
        }
        return false;
    }
};
