#include<bits/stdc++.h>
#include<stack>
class Solution {
public:
    bool isValid(string s) {
        stack <char> st;
        int n = s.length(); 

        for(int i = 0; i < n; i++) {
            // check whether this is an closing bracket
            char ch = s[i];
            if(ch == ')' || ch == '}' || ch == ']') {
                // at the top of the stack iska hi opening bracket hona chahiye 
                if(st.empty()) return false;
                else if(ch == ')') {
                    if (st.top() != '(') return false;
                    else st.pop();
                } else if(ch == '}') {
                    if (st.top() != '{') return false;
                    else st.pop();
                } else {
                    // ch = ]
                    if (st.top() != '[') return false;
                    else st.pop();

                }
            } else {
                // opening bracket so push into the stack
                st.push(ch);
            }
        }
        
        if(st.empty()) return true;
        else return false;
    }
};
