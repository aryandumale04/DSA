/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int maxDepth(TreeNode* root) {
        // base case 
        if (root == NULL) return 0;

        int leftHeight = maxDepth(root -> left);
        int rightHeight = maxDepth(root -> right);

        int height = max(leftHeight, rightHeight) + 1;
        return height;
    }




    // optimal approach  : 
        pair<int, int> diameterAndHeight(TreeNode* root) {
        // base case 
        if (root == NULL) {
            pair<int,int> p = make_pair(0,0);
            return p;
        } 

        // diameter nikalni hai ,kaise using 3 option approach 
        pair<int,int> left = diameterAndHeight(root -> left);
        pair<int, int> right = diameterAndHeight(root -> right);

        int option1 = left.first;
        int option2 = right.first;
        int option3 = left.second + right.second;

        int diameter = max(option1, max(option2, option3));
        int height = max(left.second, right.second) + 1;
        pair<int,int> ans = make_pair(diameter, height);
        return ans; 
    }
    int diameterOfBinaryTree(TreeNode* root) {
        pair<int, int> answer = diameterAndHeight(root);
        return answer.first;
    }
};
