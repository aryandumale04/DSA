/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int heightOfTree(TreeNode* root) {
        // base case 
        if (root == NULL) return 0;

        // is root ki height nikal ke leke aao 
        int leftHeight = heightOfTree(root -> left);
        int rightHeight = heightOfTree(root -> right);
        return max(leftHeight, rightHeight) + 1;
    }
    int diameterOfBinaryTree(TreeNode* root) {
        // base case 
        if (root  == NULL) return 0;

        // root null nahi hai toh is root ki diameter leke aao 
        int leftD = diameterOfBinaryTree(root -> left);
        int rightD = diameterOfBinaryTree(root -> right);
        int comboD = heightOfTree(root  -> left) + 0 + heightOfTree(root -> right);

        int max1 = max(leftD, rightD);
        int diameter = max(max1, comboD);
        return diameter;
    }
};

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    pair<int, int> diameterAndHeight(TreeNode* root) {
        // base case 
        if (root == NULL) {
            pair<int, int> p = make_pair(0, 0);
            return p;
        }

        pair<int, int> leftTreeDiameter = diameterAndHeight(root -> left);
        pair<int, int> rightTreeDiameter = diameterAndHeight(root -> right);
    
        int comboRoot = leftTreeDiameter.second + rightTreeDiameter.second;// if  + 1 then it becomes number of nodes ,but here we want diameter in terms of number of edges which is always n - 1

        pair<int, int> ans = make_pair(0,0);
        ans.first = max(leftTreeDiameter.first, max(rightTreeDiameter.first, comboRoot));
        ans.second = max(leftTreeDiameter.second, rightTreeDiameter.second) + 1; 
        return ans;
    }
    int diameterOfBinaryTree(TreeNode* root) {
        pair<int, int> answer =  diameterAndHeight(root);
        return answer.first;
    }
};
