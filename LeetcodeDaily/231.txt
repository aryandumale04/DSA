class Solution {
public:
    bool isPowerOfTwo(int n) {
        // handle negative values , 0 ,1 and posotive numbers
       
        
        for(int i = 0; i <= 30 ; i++){
            //because only uptil 2 power 30 in int range and in question asked is  INT an power of two
            int num =  pow(2,i);
            if( num == n ){
                return true;
            }
        }
        return false;
    }
};



// without termination condition
