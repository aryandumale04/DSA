class Solution {
public:
    int maximum69Number (int num) {
        string number =  to_string(num);
        for(int i = 0 ; i  < number.length(); i ++){
            if(number[i] == '6'){
                // because the left most six will have highest value and changing
                it will result in HIGHEST value change
                number[i] = '9';
                break;
            }
        }
        return stoi(number);
        
    }
};
