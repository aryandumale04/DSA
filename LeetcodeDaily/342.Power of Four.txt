#include<iostream>
#include<climits>
#include<cmath>
class Solution {
public:
    bool isPowerOfFour(int n) {

        for(int i = 0 ; ; i ++){
            if(pow(4,i) > INT_MAX){
                break;
            }
            if(pow(4,i) == n){
                return true;
            }

        }
        return false;

        
    }
};
